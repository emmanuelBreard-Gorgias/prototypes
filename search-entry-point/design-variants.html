Shop with AI — Search Bar Entry Point Variants

4 design options injected into the search bar chrome (not the results dropdown). All testable via Tampermonkey on Dawn-based Shopify stores.

Variant A Sparkle icon inside the input Minimal

A ✨ icon sits on the right side of the search input. Dormant (grey) when empty, activates (purple) as the user types. Click opens SA with the current query. Non-intrusive — doesn't change the search UX at all.

MERCHANT
✨
Shop with AI
Hypothesis Power users who are typing will notice the icon and click it.
Risk Very low discoverability. Most shoppers won't notice a grey icon.
Best for Low-disruption test. No visual footprint, minimal impact on existing UX.
Variant B Adjacent "Ask AI" pill button Prominent

A distinct pill button sits to the right of the search bar. When the user types, its label updates to reflect the query: "✨ Ask AI" → "✨ Ask: red dress". Highest discoverability without modifying the search field itself.

MERCHANT
Hypothesis Shoppers see a distinct AI option before committing to a regular search.
Risk Adds width to the search bar row. May not fit on mobile without layout changes.
Best for Highest-signal test. If this doesn't convert, nothing will.
Variant C Search / Ask AI mode toggle Structural

Two tabs at the top of the search container: "Search" and "✨ Ask AI". Clicking the AI tab changes the placeholder to "Ask anything..." and routes submission to SA. The most explicit signal that two distinct modes exist.

MERCHANT
Hypothesis Shoppers who see "Ask AI" as a peer option to Search will opt in intentionally.
Risk Most invasive injection. Adds elements above the input. May conflict with custom themes.
Best for Testing whether explicit mode choice increases intent quality (not just volume).
Variant D Persistent strip below the input Contextual

A thin strip attaches to the bottom edge of the search input (above the dropdown). Always visible when search is focused. Updates with the user's query. Visually connected to the bar, clearly separate from search results.

MERCHANT
✨ Shop with AI →
Hypothesis Shoppers who open search see AI as an escape hatch — especially when they're unsure what to type.
Risk Overlap with the dropdown. When results appear, the strip sits between the input and results, which can feel noisy.
Best for Catching the "I don't know what to search" moment — empty or partial query state.

Tampermonkey Scripts — Dawn Theme (Shopify)

Each script targets Dawn's DOM structure: input[name="q"] as the search input, .search-modal__form or .header__search as the wrapper. Click to expand each script.

Variant A — Sparkle icon inside input click to expand 
// ==UserScript==
// @name         SA Search — Variant A (Icon inside input)
// @match        https://*/*
// @grant        none
// ==/UserScript==

(function () {
  const PURPLE = '#7c3aed';

  function init() {
    const input = document.querySelector('input[name="q"]');
    if (!input) return;

    // Ensure the parent is positioned
    const wrapper = input.parentElement;
    if (getComputedStyle(wrapper).position === 'static') {
      wrapper.style.position = 'relative';
    }

    // Create icon
    const icon = document.createElement('button');
    icon.id = 'gorgias-sa-icon';
    icon.textContent = '✨';
    icon.title = 'Shop with AI';
    icon.style.cssText = `
      position: absolute;
      right: 12px;
      top: 50%;
      transform: translateY(-50%);
      background: none;
      border: none;
      cursor: pointer;
      font-size: 16px;
      padding: 4px;
      border-radius: 4px;
      color: #bbb;
      transition: color 0.15s, background 0.15s;
      line-height: 1;
      z-index: 10;
    `;

    // Adjust input padding so text doesn't go under icon
    input.style.paddingRight = '44px';

    wrapper.appendChild(icon);

    // Color icon when user types
    input.addEventListener('input', () => {
      icon.style.color = input.value ? PURPLE : '#bbb';
    });

    icon.addEventListener('mouseenter', () => {
      icon.style.background = '#f5f0ff';
      icon.style.color = PURPLE;
    });

    icon.addEventListener('mouseleave', () => {
      icon.style.background = 'none';
      icon.style.color = input.value ? PURPLE : '#bbb';
    });

    icon.addEventListener('click', (e) => {
      e.preventDefault();
      e.stopPropagation();
      openSA(input.value.trim());
    });
  }

  function openSA(query) {
    if (typeof GorgiasChat !== 'undefined') {
      GorgiasChat.open();
      GorgiasChat.setPage('chat');
      if (query) GorgiasChat.sendMessage(query);
    } else {
      console.log('[SA] Would open with query:', query || '(empty)');
    }
  }

  if (document.readyState === 'loading') {
    document.addEventListener('DOMContentLoaded', init);
  } else {
    init();
  }
})();
Variant B — Adjacent pill button click to expand 
// ==UserScript==
// @name         SA Search — Variant B (Adjacent pill)
// @match        https://*/*
// @grant        none
// ==/UserScript==

(function () {
  const PURPLE = '#7c3aed';

  function init() {
    const input = document.querySelector('input[name="q"]');
    if (!input) return;

    // Wrap input + pill in a flex row
    const wrapper = input.parentElement;
    const row = document.createElement('div');
    row.style.cssText = 'display:flex;align-items:center;gap:10px;width:100%;';

    // Insert row before input, then move input into row
    wrapper.insertBefore(row, input);
    row.appendChild(input);
    input.style.flex = '1';

    // Create pill
    const pill = document.createElement('button');
    pill.id = 'gorgias-sa-pill';
    pill.innerHTML = '✨ Ask AI';
    pill.style.cssText = `
      flex-shrink: 0;
      height: 46px;
      padding: 0 16px;
      background: white;
      border: 1.5px solid ${PURPLE};
      border-radius: 8px;
      color: ${PURPLE};
      font-size: 13px;
      font-weight: 600;
      cursor: pointer;
      display: flex;
      align-items: center;
      gap: 6px;
      white-space: nowrap;
      transition: background 0.15s, color 0.15s;
      font-family: inherit;
    `;

    row.appendChild(pill);

    const pillText = pill.querySelector('#gorgias-sa-pill-text');

    // Update pill text as user types
    input.addEventListener('input', () => {
      const q = input.value.trim();
      pillText.textContent = q ? `Ask: "${q.length > 20 ? q.slice(0, 20) + '…' : q}"` : 'Ask AI';
    });

    pill.addEventListener('mouseenter', () => {
      pill.style.background = PURPLE;
      pill.style.color = 'white';
    });

    pill.addEventListener('mouseleave', () => {
      pill.style.background = 'white';
      pill.style.color = PURPLE;
    });

    pill.addEventListener('click', (e) => {
      e.preventDefault();
      e.stopPropagation();
      openSA(input.value.trim());
    });
  }

  function openSA(query) {
    if (typeof GorgiasChat !== 'undefined') {
      GorgiasChat.open();
      GorgiasChat.setPage('chat');
      if (query) GorgiasChat.sendMessage(query);
    } else {
      console.log('[SA] Would open with query:', query || '(empty)');
    }
  }

  if (document.readyState === 'loading') {
    document.addEventListener('DOMContentLoaded', init);
  } else {
    init();
  }
})();
Variant C — Mode toggle tabs click to expand 
// ==UserScript==
// @name         SA Search — Variant C (Mode toggle)
// @match        https://*/*
// @grant        none
// ==/UserScript==

(function () {
  const PURPLE = '#7c3aed';
  let mode = 'search'; // 'search' | 'ai'

  function init() {
    const input = document.querySelector('input[name="q"]');
    if (!input) return;

    const wrapper = input.parentElement;

    // Create tab row
    const tabs = document.createElement('div');
    tabs.style.cssText = `
      display: flex;
      border-bottom: 1px solid #eee;
      margin-bottom: 0;
    `;

    const searchTab = makeTab('🔍 Search', true);
    const aiTab = makeTab('✨ Ask AI', false);

    tabs.appendChild(searchTab);
    tabs.appendChild(aiTab);

    // Wrap in container
    const container = document.createElement('div');
    container.style.cssText = `
      border: 1.5px solid #d8d8d4;
      border-radius: 10px;
      overflow: hidden;
      background: white;
    `;

    // Detach input, build container
    wrapper.insertBefore(container, input);
    container.appendChild(tabs);
    container.appendChild(input);
    input.style.cssText += 'border:none;border-radius:0;outline:none;padding:12px 14px;width:100%;display:block;';

    searchTab.addEventListener('click', () => {
      mode = 'search';
      searchTab.style.fontWeight = '700';
      searchTab.style.color = '#1a1a1a';
      searchTab.style.borderBottom = '2px solid #1a1a1a';
      searchTab.style.background = '#f8f8f6';
      aiTab.style.fontWeight = '500';
      aiTab.style.color = '#999';
      aiTab.style.borderBottom = '2px solid transparent';
      aiTab.style.background = 'none';
      input.placeholder = 'Search products...';
    });

    aiTab.addEventListener('click', () => {
      mode = 'ai';
      aiTab.style.fontWeight = '700';
      aiTab.style.color = PURPLE;
      aiTab.style.borderBottom = `2px solid ${PURPLE}`;
      aiTab.style.background = '#faf5ff';
      searchTab.style.fontWeight = '500';
      searchTab.style.color = '#999';
      searchTab.style.borderBottom = '2px solid transparent';
      searchTab.style.background = 'none';
      input.placeholder = 'Ask anything — "What\'s good for sensitive skin?"';
      input.focus();
    });

    // Intercept form submission in AI mode
    const form = input.closest('form');
    if (form) {
      form.addEventListener('submit', (e) => {
        if (mode === 'ai') {
          e.preventDefault();
          e.stopImmediatePropagation();
          openSA(input.value.trim());
        }
      });
    }

    // Also intercept Enter key
    input.addEventListener('keydown', (e) => {
      if (e.key === 'Enter' && mode === 'ai') {
        e.preventDefault();
        e.stopImmediatePropagation();
        openSA(input.value.trim());
      }
    });
  }

  function makeTab(label, active) {
    const tab = document.createElement('button');
    tab.textContent = label;
    tab.type = 'button';
    tab.style.cssText = `
      flex: 1;
      padding: 9px 12px;
      font-size: 12px;
      font-weight: ${active ? '700' : '500'};
      cursor: pointer;
      text-align: center;
      color: ${active ? '#1a1a1a' : '#999'};
      border: none;
      border-bottom: 2px solid ${active ? '#1a1a1a' : 'transparent'};
      background: ${active ? '#f8f8f6' : 'none'};
      transition: all 0.15s;
      font-family: inherit;
    `;
    return tab;
  }

  function openSA(query) {
    if (typeof GorgiasChat !== 'undefined') {
      GorgiasChat.open();
      GorgiasChat.setPage('chat');
      if (query) GorgiasChat.sendMessage(query);
    } else {
      console.log('[SA] Would open with query:', query || '(empty)');
    }
  }

  if (document.readyState === 'loading') {
    document.addEventListener('DOMContentLoaded', init);
  } else {
    init();
  }
})();
Variant D — Persistent strip below input click to expand 
// ==UserScript==
// @name         SA Search — Variant D (Strip below input)
// @match        https://*/*
// @grant        none
// ==/UserScript==

(function () {
  const PURPLE = '#7c3aed';

  function init() {
    const input = document.querySelector('input[name="q"]');
    if (!input) return;

    const wrapper = input.parentElement;

    // Create the strip
    const strip = document.createElement('div');
    strip.id = 'gorgias-sa-strip';

    const label = document.createElement('span');
    label.style.cssText = `font-size:12px;font-weight:600;color:${PURPLE};display:flex;align-items:center;gap:6px;`;
    label.innerHTML = '✨ Shop with AI';

    const query = document.createElement('span');
    query.id = 'gorgias-sa-strip-query';
    query.style.cssText = `font-size:12px;color:${PURPLE};font-weight:400;`;

    const arrow = document.createElement('span');
    arrow.textContent = '→';
    arrow.style.cssText = `margin-left:auto;color:${PURPLE};font-size:12px;`;

    strip.style.cssText = `
      display: flex;
      align-items: center;
      gap: 8px;
      padding: 9px 14px;
      background: #faf5ff;
      cursor: pointer;
      border: 1.5px solid #d8d8d4;
      border-top: none;
      border-radius: 0 0 8px 8px;
      transition: background 0.15s;
    `;

    strip.appendChild(label);
    strip.appendChild(query);
    strip.appendChild(arrow);

    // Round off the top of the input wrapper
    wrapper.style.borderRadius = '8px 8px 0 0';

    // Insert strip after the wrapper
    wrapper.parentElement.insertBefore(strip, wrapper.nextSibling);

    // Update query text as user types
    input.addEventListener('input', () => {
      const q = input.value.trim();
      query.textContent = q ? `"${q.length > 24 ? q.slice(0, 24) + '…' : q}"` : '';
    });

    strip.addEventListener('mouseenter', () => {
      strip.style.background = '#f0e8ff';
    });

    strip.addEventListener('mouseleave', () => {
      strip.style.background = '#faf5ff';
    });

    strip.addEventListener('click', (e) => {
      e.preventDefault();
      e.stopPropagation();
      openSA(input.value.trim());
    });
  }

  function openSA(query) {
    if (typeof GorgiasChat !== 'undefined') {
      GorgiasChat.open();
      GorgiasChat.setPage('chat');
      if (query) GorgiasChat.sendMessage(query);
    } else {
      console.log('[SA] Would open with query:', query || '(empty)');
    }
  }

  if (document.readyState === 'loading') {
    document.addEventListener('DOMContentLoaded', init);
  } else {
    init();
  }
})();
